export function ColorSwatch({ token }) {
  const raw = typeof document !== 'undefined'
    ? getComputedStyle(document.documentElement).getPropertyValue(token).trim()
    : '';
  // Space-separated RGB channels (e.g. "0 79 145") need wrapping in rgb().
  // Hex, named colors, oklch, etc. can be used directly.
  const isChannels = /^\d/.test(raw);
  const color = isChannels ? `rgb(${raw})` : raw || 'transparent';
  return (
    <div style={{ display: 'flex', alignItems: 'center', gap: '12px', marginBottom: '8px' }}>
      <div style={{
        width: '48px',
        height: '48px',
        borderRadius: '4px',
        background: color,
        border: '1px solid rgba(0,0,0,0.1)',
        flexShrink: 0,
      }} />
      <div>
        <code style={{ display: 'block', fontSize: '13px' }}>{token}</code>
        <span style={{ fontSize: '12px', color: '#666' }}>{raw || '(not resolved)'}</span>
      </div>
    </div>
  );
}

export function PaletteGroup({ label, tokens }) {
  return (
    <div style={{ marginBottom: '32px' }}>
      <h3 style={{ marginBottom: '16px', borderBottom: '1px solid #eee', paddingBottom: '8px' }}>{label}</h3>
      {tokens.map(t => <ColorSwatch key={t} token={t} />)}
    </div>
  );
}
