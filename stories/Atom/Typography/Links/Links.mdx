export const getCaptionForLocale = (locale) => {
  switch(locale) {
    case 'english': const engText = {detail: 'Default link'}; return engText;
    case 'arabic': const arabicText = {detail: 'الارتباط الافتراضي'}; return arabicText;
    case 'burmese': const burmeseText = {detail: 'မူရင်းလင့်ခ်'}; return burmeseText;
    case 'japanese': const japaneseText = {detail: 'デフォルトのリンク'}; return japaneseText;
    default:
      return {detail: 'Default link'};
  }
};

# Link

The Link component is used to attach links to the text component.

### Overview

The Link component defines a hyperlink that is used to link one page to another.

#### When to use:

When hyperlinks are to be attached.

### Formatting

#### Default

The Link component consists of the link with a red horizontal line below it.

### Behaviors

#### States

There are three states in the Link component: Desktop and Laptop. The component is compatible with tablets and mobile.

Desktop and Laptop: The font size of the Link is biggest in Desktop and Laptop.
