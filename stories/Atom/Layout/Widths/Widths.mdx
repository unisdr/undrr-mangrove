# Widths

> **@deprecated** — Legacy from the UNDP design system. These width tokens are not part of the current UNDRR distribution and are not used by any active Mangrove components. They will be removed in a future major release. Use standard CSS layout techniques (flexbox, grid) or Mangrove's `$mg-spacing-*` and `$mg-breakpoint-*` tokens instead.

export const WIDTH_TOKENS = [
  '--mg-width-150',
  '--mg-width-300',
  '--mg-width-400',
  '--mg-width-600',
];

export function WidthRow({ token }) {
  const value = typeof document !== 'undefined'
    ? getComputedStyle(document.documentElement).getPropertyValue(token).trim()
    : '';
  return (
    <tr>
      <td style={{ padding: '6px 12px', fontFamily: 'monospace', fontSize: '13px' }}>{token}</td>
      <td style={{ padding: '6px 12px', fontSize: '13px' }}>{value || '—'}</td>
      <td style={{ padding: '6px 12px' }}>
        <div style={{ width: value || 0, height: '16px', background: 'var(--mg-color-interactive, #004f91)', borderRadius: '2px' }} />
      </td>
    </tr>
  );
}
