export const getCaptionForLocale = locale => {
  switch (locale) {
    case 'english':
      const engText = { detail: 'Read more' };
      return engText;
    case 'arabic':
      const arabicText = { detail: 'اقرأ أكثر' };
      return arabicText;
    case 'japanese':
      const japaneseText = { detail: '続きを読む' };
      return japaneseText;
    default:
      return { detail: 'Read more' };
  }
};

# Call-to-action buttons

Primary and secondary action buttons used to prompt users to take an action like "Read more" or "Sign up."

#### When to use:

- Buttons should be used as primary, secondary or tertiary calls to actions
- Ideally, a page should have one primary action. A secondary/tertiary option should be used infrequently.
- If there are multiple choices for an action item, one default button should be active and the other buttons inactive

### Formatting

#### Default

There are multiple button types in our design system

1. Primary/Secondary/tertiary buttons
2. Chips

### Content

Button and chip colors and text indicate the default setting and call to action.
